96 baylib::marginal_distribution<probability_type> make_inference ()
97 {
98 BAYLIB_ASSERT(std::all_of(bn.begin(), bn.end(),
99 [this](auto &var){ return baylib::cpt_filled_out(bn, var.id()); }),
100 "conditional probability tables must be properly filled to"
101 " run logic_sampling inference algorithm",
102 std::runtime_error);
103
104 icpt_vector icptvec{};
105 auto result = marginal_distribution<probability_type>(bn.begin(), bn.end());
106 bool evidence_found = false;
107 for (int v_id = 0; v_id < bn.number_of_variables(); ++v_id){
108 icptvec.emplace_back(cow::icpt<probability_type>(bn[v_id].table()));
109 if(bn[v_id].is_evidence()){
110 result[v_id][bn[v_id].evidence_state()] = 1;
111 evidence_found = true;
112 }
113 }
114 // If no evidence is present the algorithm degenerates to simple
115 // logic_sampling, and we can skip the learning phase
116 if(evidence_found){
117 ancestors = ancestors_of_evidence(bn);
118 learn_icpt(icptvec);
119 }
120 result += gpu_simulation(icptvec, bn);
121 result.normalize();
122 return result;
123 }

149 void learn_icpt( icpt_vector & icptvec )
150 {
151 ulong nvars = bn.number_of_variables();
152 simulation_matrix graph_state(learning_step);
153 std::vector<probability_type> random_vec(learning_step * nvars);
154 double k = 0;
155 uint max_k = this->nsamples / 2 / learning_step;
156 seed_factory factory(1, this->seed);
157 baylib::random_generator<probability_type, Generator_> rnd_gen(this->seed);
158
159 for (int i = 0; i < max_k; ++i) {
160 std::future<void> el = std::async([&](){std::generate(
161 random_vec.begin(),
162 random_vec.end(),
163 rnd_gen);});
164 if(i != 0){
165 // Formula for the learning_rate proposed in the paper
166 double learning_rate = initial_learning_rate *
167 std::pow(final_learning_rate / initial_learning_rate,k / (double)max_k);
168 double difference = absorb_samples(graph_state, bn, icptvec, learning_rate);
169 // if the maximum difference was low enough we stop the learning process
170 if(difference < learning_cutoff){
171 break;
172 }
173 k++;
174 }
175 el.get();
176
177 std::transform( std::execution::par,
178 boost::counting_iterator<ulong>(0),
179 boost::counting_iterator<ulong>(graph_state.size()),
180 graph_state.begin(),
181 [&](ulong ix)
182 {
183 std::vector<uint> local_result(nvars);
184 ix *= nvars;
185 for (ulong v: ancestors) {
186 if (bn[v].is_evidence()) {
187 local_result[v] = bn[v].evidence_state();
188 continue;
189 }
190 const probability_type p = random_vec[ix];
191 ix++;
192 std::vector<probability_type> weight;
193 baylib::condition parents_state_cond;
194 for (auto par : bn.parents_of(v))
195 parents_state_cond.add(
196 par,
197 local_result[par]
198 );
199 weight = icptvec[v][parents_state_cond];
200 ulong sample = make_random_by_weight(p, weight);
201 local_result[v] = sample;
202 }
203 return local_result;
204 });
205 }
206 this->seed = factory.get_new(); // new seed for gpu simulation
207 }

218 double absorb_samples(
219 const simulation_matrix & graph_state,
220 const network_type & bn,
221 icpt_vector & icptvec,
222 double learning_rate
223 )
224 {
225 probability_type evidence_score;
226 std::vector<probability_type> sample_weight(graph_state.size());
227
228 double max_distance = 0.;
229
230 // Calculate the likelihood of extracting a particular set of samples
231 std::transform(std::execution::par_unseq,
232 boost::counting_iterator<uint>(0),
233 boost::counting_iterator<uint>(graph_state.size()),
234 sample_weight.begin(),
235 [&](uint ix)
236 {
237 probability_type weight = 1;
238 for(ulong v_id: ancestors){
239 condition cond;
240 auto& icpt = icptvec[v_id];
241 auto& cpt = bn[v_id].table();
242 auto& sample_state = graph_state[ix][v_id];
243
244 for(auto p_id : bn.parents_of(v_id))
245 cond.add(p_id, graph_state[ix][p_id]);
246 if(bn[v_id].is_evidence()){
247 weight *= cpt[cond][bn[v_id].evidence_state()];
248 }else{
249 weight *= cpt[cond][sample_state] / icpt[cond][sample_state];
250 }
251 }
252 return weight;});
253
254 // Update the icpts and return the maximum distance
255 max_distance = std::transform_reduce(
256 std::execution::par_unseq,
257 ancestors.begin(),
258 ancestors.end(),
259 0.,
260 [](auto e1, auto e2){return e1 > e2 ? e1 : e2;},
261 [&](auto v_id)
262 {
263 if(bn[v_id].is_evidence())
264 return 0.;
265 auto& original_cpt = icptvec[v_id];
266 cow::icpt<probability_type> temp_icpt(bn[v_id].table(), true);
267 for (int i = 0; i < graph_state.size(); ++i) {
268 condition cond;
269 auto sample = graph_state[i][v_id];
270 for(auto p_id : bn.parents_of(v_id))
271 cond.add(p_id, graph_state[i][p_id]);
272 temp_icpt[cond][sample] += sample_weight[i];
273 }
274 temp_icpt.normalize();
275 double distance = original_cpt.absorb(temp_icpt, learning_rate);
276 return distance;
277 });
278
279 return max_distance;
280 }

289 marginal_distribution<probability_type> gpu_simulation(
290 const icpt_vector& icpt_vec,
291 const network_type & bn
292 )
293 {
294 int niter = 1;
295 marginal_distribution<probability_type> marginal_result(bn.begin(), bn.end());
296 std::vector<bcvec> result_container(bn.number_of_variables());
297 marginal_distribution<probability_type> temp(bn.begin(), bn.end());
298 auto [gpu_samples, gpu_iter] = this->calculate_iterations();
299
300 for(int i = 0; i < gpu_iter; ++i){
301 for(ulong v : sampling_order(bn)) {
302 if(bn[v].is_evidence()){
303 result_container[v] = bcvec(this->nsamples, bn[v].number_of_states(), this->context);
304 compute::fill(result_container[v].state.begin(), result_container[v].state.end(), bn[v].evidence_state(), this->queue);
305 }
306 else{
307 std::vector<bcvec*> parents_result;
308 auto parents = bn.parents_of(v);
309
310 for (auto p : parents) {
311 parents_result.push_back(&result_container[p]);
312 }
313
314 result_container[v] = this->simulate_node(v, icpt_vec[v], parents_result, gpu_samples);
315
316 auto accumulated_result = compute_result_general(result_container[v]);
317
318 for(int ix=0; ix< accumulated_result.size(); ix++)
319 marginal_result[v][ix] += accumulated_result[ix];
320 }
321 }
322 }
323 return marginal_result;
324 }

331 std::vector<ulong> compute_result_general(bcvec& res)
332 {
333 std::vector<ulong> acc_res(res.cardinality);
334 for (baylib::state_t i = 0; i < res.cardinality; ++i) {
335 acc_res[i] = w_k * compute::count(res.state.begin(), res.state.end(), i, this->queue);
336 }
337 return acc_res;
338 }

347 uint make_random_by_weight(
348 const probability_type p,
349 const std::vector<probability_type> & weight
350 )
351 {
352 probability_type total = 0.0;
353 for(uint i = 0; i < weight.size(); ++i)
354 {
355 auto const old_total = total;
356 total += weight[i];
357 if(old_total <= p && p < total)
358 {
359 return i;
360 }
361 }
362 return weight.size() - 1;
363 }
364 };
