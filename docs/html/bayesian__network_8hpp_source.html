429 void optimize_cpt_memory_occupation(vertex_id id){
430 auto seed = variable(id).cpt.hash();
431 if(cpt_hash_map.find(seed) != cpt_hash_map.end()){
432 auto var = variable(cpt_hash_map[seed]);
433 if(var._id != id && var.cpt == variable(id).cpt){
434 variable(id).cpt.d = var.cpt.d;
