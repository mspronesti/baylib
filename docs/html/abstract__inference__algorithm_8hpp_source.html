238 bcvec simulate_node(
239 ulong v_id,

242 int dim
243 )
244 {
245 std::vector<probability_type> flat_cpt_accum = accumulate_cpt(v_id, cpt);
246 bcvec result(dim, cpt.number_of_states(), context);
247 prob_v device_cpt(flat_cpt_accum.size(), context);
248 prob_v threshold_vec(dim, context);
249 prob_v random_vec(dim, context);
250 compute::uniform_real_distribution<probability_type> distribution(0, 1);
251 compute::vector<int> index_vec(dim, context);
252
253 // Async copy of the cpt in gpu memory
254 compute::copy(flat_cpt_accum.begin(), flat_cpt_accum.end(), device_cpt.begin(), queue);
255
256 // cycle for deducing the row of the cpt given the parents state in the previous simulation
257 if(parents_result.empty())
258 compute::fill(index_vec.begin(), index_vec.end(), 0, queue);
259 else {
260 uint coeff = bn[v_id].table().number_of_states();
261 for (int i = 0; i < parents_result.size(); i++) {
262 if (i == 0)
263 compute::transform(parents_result[i]->state.begin(),
264 parents_result[i]->state.end(),
265 index_vec.begin(),
266 _1 * coeff, queue);
267 else
268 compute::transform(parents_result[i]->state.begin(),
269 parents_result[i]->state.end(),
270 index_vec.begin(),
271 index_vec.begin(),
272 _1 * coeff + _2, queue);
273 coeff *= parents_result[i]->cardinality;
274 }
275 }
276
277 // get the threshold corresponding to the specific row of the cpt for every single simulation
278 compute::gather(index_vec.begin(),
279 index_vec.end(),
280 device_cpt.begin(),
281 threshold_vec.begin(), queue);
282
283
284 // generate random vector
285 distribution.generate(random_vec.begin(),
286 random_vec.end(),
287 rand, queue);
288
289 // confront the random vector with the threshold
290 compute::transform(random_vec.begin(),
291 random_vec.end(),
292 threshold_vec.begin(),
293 result.state.begin(),
294 _1 > _2,
295 queue);
296
297 // generalization in case of more than 2 states
298 for (int i = 0; i + 2 < bn[v_id].table().number_of_states(); i++) {
299 compute::vector<int> temp(dim, context);
300 compute::transform(index_vec.begin(),
301 index_vec.end(),
302 index_vec.begin(),
303 _1 + 1, queue);

309 random_vec.end(),

311 temp.begin(),
312 _1 > _2, queue);
313 compute::transform(temp.begin(),

315 result.state.begin(),
316 result.state.begin(),
317 _1 + _2, queue);
