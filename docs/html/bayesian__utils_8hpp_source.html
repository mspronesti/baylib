37 template <BNetDerived Network_>
38 std::vector<unsigned long> sampling_order (
39 const Network_ &bn
40 )
41 {
42 ulong nvars = bn.number_of_variables();
43
44 // initially, the all have rank 0
45 auto ranks = std::vector<unsigned long>(nvars, 0);
46 auto roots = std::vector<unsigned long>{};
47
48 for(ulong vid = 0; vid < nvars; ++vid)
49 if(bn.is_root(vid))
50 roots.push_back(vid);
51
52 BAYLIB_ASSERT(!roots.empty(),
53 "No root vertices found in graph",
54 std::runtime_error)
55
56 while(!roots.empty()) {
57 unsigned long curr_node = roots.back();
58 roots.pop_back();
59
60 for(ulong vid = 0; vid < nvars; ++vid) {
61 if (!bn.has_dependency(curr_node, vid)) continue;
62
63 if (ranks[curr_node] + 1 > ranks[vid]) {
64 ranks[vid] = ranks[curr_node] + 1;
65 roots.push_back(vid);
66 }
67 }
68 }
69
70 auto order = std::vector<unsigned long>(nvars);
71 std::iota(order.begin(), order.end(), 0);
72
73 std::sort(order.begin(), order.end(), [&ranks](auto &a, auto&b){
74 return ranks[a] < ranks[b];
75 });
76
77 return order;
78 }

88 template <BNetDerived Network_>
89 std::unordered_set<unsigned long> markov_blanket (
90 const Network_ &bn,
91 unsigned long vid
92 )
93 {
94 auto marblank = std::unordered_set<unsigned long>{};
95 for(auto & pv : bn.parents_of(vid))
96 marblank.insert(pv);
97
98 for(auto & v : bn.children_of(vid)){
99 marblank.insert(vid);
100 for(auto pv : bn.parents_of(v))
101 if(v != vid)
102 marblank.insert(pv);
103 }
104
105 return marblank;
106 }

116 template <BNetDerived Network_>
117 bool cpt_filled_out(
118 const Network_ &bn,
119 const unsigned long cpt_owner
120 )
121 {
122 bn::condition_factory factory(bn, cpt_owner);
123 const auto &cpt = bn[cpt_owner].table();
124
125 if(factory.number_of_combinations() != cpt.size())
126 return false;
127
128 do {
129 auto cond = factory.get();
130 if(!cpt.has_entry_for(cond))
131 return false;
132 } while(factory.has_next());
133
134 for(auto & row : cpt) {
135 auto sum = std::accumulate(row.begin(), row.end(), 0.0);
136 if (abs(sum - 1.0) > 1.0e-5) {
137 return false;
138 }
139 }
140
141 return true;
142 }

165 template <BNetDerived Network_>
166 std::vector<ulong> ancestors_of_evidence(const Network_ &bn){
167 std::vector<bool> ancestor(bn.number_of_variables(), false);
168 std::function<void(ulong)> mark_ancestors;
169 mark_ancestors = [&bn, &ancestor, &mark_ancestors](ulong v_id){
170 ancestor[v_id] = true;
171 for (ulong p_id: bn.parents_of(v_id)) {
172 if(ancestor[p_id])
173 continue;
174 ancestor[p_id] = true;
175 mark_ancestors(p_id);
176 }
177 };
178 for (uint i = 0; i < bn.number_of_variables(); ++i) {
179 if(bn[i].is_evidence())
180 mark_ancestors(i);
181 }
182 std::vector<ulong> ordered_result;
183 for (auto vertex: sampling_order(bn)) {
184 if(ancestor[vertex])
185 ordered_result.emplace_back(vertex);
186 }
187 return ordered_result;
188 }
