39 template <typename Probability>
40 std::vector<bn::vertex<Probability>> sampling_order(const bn::bayesian_network<Probability> &bn){
41 using vertex_t = bn::vertex<Probability>;
42 ulong nvars = bn.number_of_variables();
43
44 // initially, the all have rank 0
45 auto ranks = std::vector<vertex_t>(nvars, 0);
46 auto roots = std::vector<vertex_t>{};
47
48 for(ulong vid = 0; vid < nvars; ++vid)
49 if(bn.is_root(vid))
50 roots.push_back(vid);
51
52 BAYLIB_ASSERT(!roots.empty(),
53 "No root vertices found in graph",
54 std::runtime_error)
55
56 while(!roots.empty()) {
57 vertex_t curr_node = roots.back();
58 roots.pop_back();
59
60 for(ulong vid = 0; vid < nvars; ++vid) {
61 if (!bn.has_dependency(curr_node, vid)) continue;
62
63 if (ranks[curr_node] + 1 > ranks[vid]) {
64 ranks[vid] = ranks[curr_node] + 1;
65 roots.push_back(vid);
66 }
67 }
68 }
69
70 auto order = std::vector<vertex_t>(nvars);
71 std::iota(order.begin(), order.end(), 0);
72
73 std::sort(order.begin(), order.end(), [&ranks](auto &a, auto&b){
74 return ranks[a] < ranks[b];
75 });
76
77 return order;
78 }

89 std::unordered_set<bn::vertex<Probability>> markov_blanket
90 ( const bn::bayesian_network<Probability> &bn,
91 const bn::random_variable<Probability> &rv
92 ) {
93 auto rv_id = rv.id();
94 auto marblank = std::unordered_set<bn::vertex<Probability>>{};
95 for(auto & pv : bn.parents_of(rv_id))
96 marblank.insert(pv);
97
98 for(auto & vid : bn.children_of(rv_id)){
99 marblank.insert(vid);
100 for(auto pv : bn.parents_of(vid))
101 if(vid != rv_id)
102 marblank.insert(pv);
103 }
104
105 return marblank;

116 template <typename Probability>
117 bool cpt_filled_out(const bn::random_variable<Probability> &cpt_owner)
118 {
119 bn::condition_factory factory(cpt_owner);
120 const auto &cpt = cpt_owner.table();
121
122 if(factory.number_of_combinations() != cpt.size())
123 return false;
124
125 do {
126 auto cond = factory.get();
127 if(!cpt.has_entry_for(cond))
128 return false;
129 } while(factory.has_next());
130
131 for(auto & row : cpt) {
132 Probability sum = std::accumulate(row.begin(), row.end(), 0.0);
133 if (abs(sum - 1.0) > 1.0e-5) {
134 return false;
135 }
136 }
137
138 return true;
139 }

162 template <typename Probability>
163 std::vector<ulong> ancestors_of_evidence(const bn::bayesian_network<Probability> &bn){
164 std::vector<bool> ancestor(bn.number_of_variables(), false);
165 std::function<void(ulong)> mark_ancestors;
166 mark_ancestors = [&bn, &ancestor, &mark_ancestors](ulong v_id){
167 ancestor[v_id] = true;
168 for (ulong p_id: bn.parents_of(v_id)) {
169 if(ancestor[p_id])
170 continue;
171 ancestor[p_id] = true;
172 mark_ancestors(p_id);
173 }
174 };
175 for (uint i = 0; i < bn.number_of_variables(); ++i) {
176 if(bn[i].is_evidence())
177 mark_ancestors(i);
178 }
179 std::vector<ulong> ordered_result;
180 for (auto vertex: sampling_order(bn)) {
181 if(ancestor[vertex])
182 ordered_result.emplace_back(vertex);
183 }
184 return ordered_result;
185 }
