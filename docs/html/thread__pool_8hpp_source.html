125 template <typename T, typename F>
126 void parallelize_loop(T first_index, T last_index, const F &loop, ui32 num_tasks = 0)
127 {
128 if (num_tasks == 0)
129 num_tasks = thread_count;
130 if (last_index < first_index)
131 std::swap(last_index, first_index);
132 size_t total_size = last_index - first_index + 1;
133 size_t block_size = total_size / num_tasks;
134 if (block_size == 0)
135 {
136 block_size = 1;
137 num_tasks = (ui32)total_size > 1 ? (ui32)total_size : 1;
138 }
139 std::atomic<ui32> blocks_running = 0;
140 for (ui32 t = 0; t < num_tasks; t++)
141 {
142 T start = (T)(t * block_size + first_index);
143 T end = (t == num_tasks - 1) ? last_index : (T)((t + 1) * block_size + first_index - 1);
144 blocks_running++;
145 push_task([start, end, &loop, &blocks_running]
146 {
147 for (T i = start; i <= end; i++)
148 loop(i);
149 blocks_running--;
150 });
151 }
152 while (blocks_running != 0)
153 {
154 sleep_or_yield();
155 }
156 }
