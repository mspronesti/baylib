68 auto [iter_samples, niter] = this->calculate_iterations(bn);
69 auto vertex_queue = bn::sampling_order(bn);
70 marginal_distribution<Probability_> marginal_result(bn.begin(), bn.end());
71 for (ulong i = 0; i< niter; i++) {
72
73 std::vector<bcvec> result_container(vertex_queue.size());
74 marginal_distribution<Probability_> temp(bn.begin(), bn.end());
75 compute::vector<int> valid_evidence_vec(this->nsamples, true, this->queue);
76
77 for(ulong v : vertex_queue) {
78
79 std::vector<bcvec*> parents_result;
80
81 // Build parents result vector in the correct order
82 auto parents = bn[v].parents_info.names();
83 std::reverse(parents.begin(), parents.end());
84 for (auto p : parents) {
85 parents_result.push_back(&result_container[bn.index_of(p)]);
86 }
87
88 result_container[v] = this->simulate_node(bn[v].table(), parents_result, iter_samples);
89
90 if(bn[v].is_evidence()){
91 compute::transform( result_container[v].state.begin()
92 ,result_container[v].state.end()
93 ,valid_evidence_vec.begin()
94 ,valid_evidence_vec.begin()
95 ,(_1 == bn[v].evidence_state()) && _2
96 ,this->queue);
97 }
98 }
99
100 for(ulong v : vertex_queue) {
101 auto accumulated_result = compute_result_general(result_container[v], valid_evidence_vec);
102 for(int ix=0; ix< accumulated_result.size(); ix++)
103 marginal_result[v][ix] += accumulated_result[ix];
104 }
105 }

108 }
