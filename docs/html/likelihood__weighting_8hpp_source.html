53 bn::marginal_distribution<probability_type> mdistr(bn.begin(), bn.end());
54 bn::random_generator<probability_type, Generator_> rnd_gen(seed);

56 for(ulong i=0; i<nsamples; i++){

58 ulong vid = 0;
59 auto weight = sample_pair.second;
60
61 for(auto & samp : sample_pair.first)

67 std::pair<pattern_t , probability_type> weighted_sample(
68 bn::random_generator<probability_type> & rnd_gen
69 )
70 {
71
72 probability_type weight = 1.0;
73 auto pattern = pattern_t(bn.number_of_variables(), 0.0);
74
75 for(ulong vid = 0; vid < bn.number_of_variables(); ++vid)
76 {
77 auto & var = bn[vid];
78 bn::condition parent_state;
79
80 for(auto par : bn.parents_of(vid))
81 parent_state.add(
82 par,
83 pattern[par]
84 );
85
86 const auto & cpt = var.table();
87 if(var.is_evidence()) {
88 ulong evidence_state = var.evidence_state();
89 weight *= cpt[parent_state][evidence_state];
90 pattern[vid] = evidence_state;
91 } else {
92 pattern[vid] = make_random_by_weight(
93 rnd_gen.get_random(),
94 cpt[parent_state]
95 );
96 }
97 }
98
99 return std::make_pair(pattern, weight);
100 }

102 uint make_random_by_weight(
103 const probability_type p,
104 const std::vector<probability_type> & weight
105 )
106 {
107 BAYLIB_ASSERT(0.0 <= p && p <= 1.0,
108 "Invalid probability value"
109 " not included in [0,1]",
110 std::logic_error);
111
112 probability_type total = 0.0;
113 for(uint i = 0; i < weight.size(); ++i)
114 {
115 auto const old_total = total;
116 total += weight[i];
117 if(old_total <= p && p < total)
118 {
119 return i;
120 }
121 }
122
123 return weight.size() - 1;
124 }
