42 bn::marginal_distribution<Probability> sample_step(
43 const bn::bayesian_network<Probability> &bn,
44 ulong nsamples,
45 uint seed
46 ) override
47 {
48 bn::marginal_distribution<Probability> mdistr(bn.begin(), bn.end());
49 bn::random_generator<Probability, Generator> rnd_gen(seed);
50
51 for(ulong i=0; i<nsamples; i++){
52 auto sample_pair = weighted_sample(bn, rnd_gen);
53 ulong vid = 0;
54 auto weight = sample_pair.second;
55
56 for(auto & samp : sample_pair.first)
57 mdistr[vid++][samp] += weight ;
58 }
59 return mdistr;
60 }

63 std::pair<pattern_t , Probability> weighted_sample(
64 const bn::bayesian_network<Probability> &bn,
65 bn::random_generator<Probability> & rnd_gen
66 )
67 {
68
69 Probability weight = 1.0;
70 auto pattern = pattern_t(bn.number_of_variables(), 0.0);
71
72 for(ulong vid = 0; vid < bn.number_of_variables(); ++vid)
73 {
74 auto & var = bn[vid];
75 bn::condition parent_state;
76
77 for(auto par : bn.parents_of(vid))
78 parent_state.add(
79 bn[par].name(),
80 pattern[par]
81 );
82
83 const auto & cpt = var.table();
84 if(var.is_evidence()) {
85 ulong evidence_state = var.evidence_state();
86 weight *= cpt[parent_state][evidence_state];
87 pattern[vid] = evidence_state;
88 } else {
89 pattern[vid] = make_random_by_weight(
90 rnd_gen.get_random(),
91 cpt[parent_state]
92 );
93 }
94 }
95
96 return std::make_pair(pattern, weight);
97 }

99 uint make_random_by_weight(
100 const Probability p,
101 const std::vector<Probability> & weight
102 )
103 {
104 BAYLIB_ASSERT(0.0 <= p && p <= 1.0,
105 "Invalid probability value"
106 " not included in [0,1]",
107 std::logic_error);
108
109 Probability total = 0.0;
110 for(uint i = 0; i < weight.size(); ++i)
111 {
112 auto const old_total = total;
113 total += weight[i];
114 if(old_total <= p && p < total)
115 {
116 return i;
117 }
118 }
119
120 return weight.size() - 1;
121 }
